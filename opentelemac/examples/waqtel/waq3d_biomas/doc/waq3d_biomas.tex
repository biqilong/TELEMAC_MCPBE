%     This part specifies the physical parameters

\subsection{Physical parameters}
%
The BIOMASS module is activated by setting \telkey{WATER QUALITY PROCESS} = 3
in the \telemac{3d} \telkey{STEERING FILE}.
%
Only the following water quality parameters have been changed
in the \waqtel \telkey{STEERING FILE} compared to the default values:
\begin{itemize}
\item \telkey{METHOD OF COMPUTATION OF RAY EXTINCTION COEFFICIENT} = 2 (Moss formula, Secchi depth is unknown),
\item \telkey{VEGETAL TURBIDITY COEFFICIENT WITHOUT PHYTO} = 0.005~m$^{-1}$,
\item \telkey{RATE OF TRANSFORMATION OF NOR TO NO3} = 0.35~d$^{-1}$,
\item \telkey{SUNSHINE FLUX DENSITY ON WATER SURFACE} = 0.01~W/m$^2$,
\item \telkey{WATER TEMPERATURE} = 20$^\circ$C (which is the mean temperature of water).
\end{itemize}

\subsection{Initial and Boundary Conditions}
%
\subsubsection{Initial conditions}
%
Initial depth~=~2~m
and no initial velocity\\
Initial values for tracers:\\
([PHY],[PO4],[POR],[NO3],[NOR]) =
(50~$\mu$gChlA/l, 3~mg/l, 0.01~mg/l, 2.9~mg/l, 9~mg/l)
%
\subsubsection{Boundary conditions}
%
Closed boundaries. No bottom friction
%
\subsection{General parameters}
%
Time step: 3,600~s =~1~h\\
Simulation duration: 8,760~h =~365~days
%
% - Numerical parameters:
%     This part is used to specify the numerical parameters used
%     (adaptive time step, mass-lumping when necessary...)
%
%
\subsection{Numerical parameters}
%
Basin at rest (no advection nor diffusion for hydrodynamics).\\
%
2 old default values for the advection and diffusion of the tracers are kept for non regression
in the \telemac{3d} \telkey{STEERING FILE} :
\begin{itemize}
\item \telkey{ACCURACY FOR DIFFUSION OF TRACERS} = 10$^{-6}$ (current default = 10$^{-8}$),
\item \telkey{SCHEME OPTION FOR ADVECTION OF TRACERS} = 1 (i.e. explicit).
\end{itemize}
